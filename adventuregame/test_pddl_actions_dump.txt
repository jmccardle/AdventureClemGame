(:action DUMP
    :parameters (?b - bucket ?t - container ?p - player ?r - room)
    :precondition (and
        (at ?p ?r)
        (at ?b ?r)
        (at ?t ?r)
        )
    :effect (and
        (when
            (type ?b plasmbucket)
            (and
                (type ectoplasm1 ectoplasm)
                (at ectoplasm1 ?r)
                (in ectoplasm1 ?t)
                (accessible ectoplasm1)
                )
            )
        (when
            (type ?b waterbucket)
            (and
                (type water1 water)
                (at water1 ?r)
                (in water1 ?t)
                (accessible water1)
                )
            )
        )
)
