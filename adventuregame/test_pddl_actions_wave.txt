(:action WAVE
    :parameters (?w - wand ?t - container ?p - player ?r - room)
    :precondition (and
        (at ?p ?r)
        (at ?w ?r)
        (at ?t ?r)
        )
    :effect (and
        (when
            (type ?w firewand)
            (and
                (not (cold ?t))
                (hot ?t)
                )
            )
        (when
            (type ?w icewand)
            (and
                (not (hot ?t))
                (cold ?t)
                )
            )
        (when
            (type ?w fairywand)
            (glittery ?t)
            )
        (when
            (type ?w zulpowand)
            (zulponated ?t)
            )
        )
)
