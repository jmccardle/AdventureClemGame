(:action TAKE
    :parameters (?e - takeable ?s - receptacle ?p - player ?i - inventory ?r - room)
    :precondition (and
        (at ?p ?r)
        (at ?e ?r)
        (not (in ?e inventory))
        (at ?s ?r)
        (accessible ?e)
        (< (itemcount ?i) 2)
        (or
            (and
                (on ?e ?s)
                (support ?s)
                )
            (and
                (open ?s)
                (container ?s)
                (in ?e ?s)
                )
            )
        )
    :effect (and
        (in ?e ?i)
        (increase (itemcount ?i) 1)
        (when
            (and
                (support ?s)
                (on ?e ?s)
                )
            (not (on ?e ?s))
            )
        (when
            (and
                (container ?s)
                (in ?e ?s)
                )
            (not (in ?e ?s))
            )
    )
)
