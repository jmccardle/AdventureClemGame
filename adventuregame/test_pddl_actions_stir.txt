(:action STIR
    :parameters (?s - stirrer ?l - potion ?p - player ?r - room)
    :precondition (and
        (at ?p ?r)
        (at ?s ?r)
        (at ?l ?r)
        )
    :effect (and
        (when
            (type ?s ladle)
            (ladlestirred ?l)
            )
        (when
            (type ?s whisk)
            (whisked ?l)
            )
        (when
            (type ?s spoon)
            (spoonstirred ?l)
            )
        (when
            (type ?s borpulus)
            (borpulusstirred ?l)
            )
        )
)
