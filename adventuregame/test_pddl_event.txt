(:event TOAD-EATS-BEETLE
    :parameters (?t - toad ?b - beetle ?r - room ?c - container)
    :precondition (and
        (at ?t ?r)
        (at ?b ?r)
        (or
            (not (in ?b ?c))
            (and
                (at ?c ?r)
                (in ?b ?c)
                (open ?c)
                )
            )
        (not (in ?b inventory))
        (hungry ?t)
        )
    :effect (and
        (not (at ?b ?r))
        (not (hungry ?t))
        )
    )
