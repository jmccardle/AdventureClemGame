// turning ectoplasm into liquid1
(:event POTIONSTEPONE
    :parameters (?l - liquid ?c - container ?r - room)
    :precondition (and
        (at ?l ?r)
        (at ?c ?r)
        (in ?l ?c)
        (type ?c cauldron)
        (type ?l ectoplasm)
        (cold ?c)
        )
    :effect (and
        (not (at ?l ?r))
        (not (in ?l ?c))
        (not (accessible ?l))
        (type liquid1 liquid)
        (at liquid1 ?r)
        (in liquid1 ?c)
        (accessible liquid1)
        )
    )

// nightshade absorbing into liquid1 yielding liquid2
(:event POTIONSTEPTWO
    :parameters (?l - liquid ?i - ingredient ?c - container ?r - room)
    :precondition (and
        (at liquid1 ?r)
        (at ?c ?r)
        (at ?i ?r)
        (type ?c cauldron)
        (cold ?c)
        (type ?i nightshade)
        (in ?i ?c)
        (in liquid1 ?c)
        )
    :effect (and
        (not (type liquid1 liquid))
        (not (at liquid1 ?r))
        (not (in liquid1 ?c))
        (not (accessible liquid1))

        (not (at ?i ?r))
        (not (in ?i ?c))
        (not (type ?i nightshade))

        (type liquid2 liquid)
        (at liquid2 ?r)
        (in liquid2 ?c)
        (accessible liquid2)
        )
    )
