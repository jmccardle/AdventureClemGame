(:action PUT
    :parameters (?e - movable ?t - receptacle ?p - player ?i - inventory ?r - room)
    :precondition (and
        (at ?p ?r)
        (at ?e ?r)
        (at ?t ?r)
        (or
            (and
                (container ?t)
                (open ?t)
                )
            (support ?t)
            )
        )
    :effect (and
        (when
            (in ?e ?i)
            (decrease (itemcount ?i) 1)
            )
        (not (in ?e ?i))
        (when
            (support ?t)
            (on ?e ?t)
            )
        (when
            (container ?t)
            (in ?e ?t)
            )
        )
)
