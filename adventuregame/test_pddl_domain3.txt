(define
    (domain home)
    (:types
        kitchen pantry hallway livingroom broomcloset bedroom - room
        player inventory floor table sidetable counter refrigerator cupboard wardrobe shelf freezer pottedplant chair bed couch broom mop sandwich apple banana orange peach plate book pillow - entity
        inventory floor table sidetable counter refrigerator cupboard wardrobe shelf freezer chair bed couch - receptacle
        )
    (:functions
        (itemcount ?i - inventory)
        )
    (:event INVENTORY-FULL
        :parameters (?i - inventory)
        :precondition (and
            (>= (itemcount ?i) 2)
            )
        :effect (and
            (full ?i)
            )
    )
    (:event INVENTORY-FREE
        :parameters (?i - inventory)
        :precondition (and
            (< (itemcount ?i) 2)
            )
        :effect (and
            (not (full ?i))
            )
    )
)
